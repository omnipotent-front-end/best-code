控制动画暂停和播放逻辑 #

通过style.animationPlayState来控制：

    let elem = document.querySelector('.animation');
	let state = elem.style['animationPlayState'];
	  
	if(state === 'paused') {
	    elem.style['animationPlayState'] = 'running';
	    btn.innerText = 'stop';
	} else {
	    elem.style['animationPlayState'] = 'paused';
	    btn.innerText = 'play';
	}
